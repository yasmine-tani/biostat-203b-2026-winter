  # hw3 labs that we want
  filter(itemid %in% c(50912, 50971, 50983, 50902, 50882, 51221, 51301, 50931)) %>%
  # keep only subjects in icustays_tble
  semi_join(icustays_tble, by = "subject_id") %>%
  # join with icustays to get the intime

    icustays_tble %>% select(subject_id, stay_id, intime),

  # labs taken BEFORE ICU admission
